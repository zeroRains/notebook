# iota

与const来表示枚举，可在const()内添加一个关键字，每行的iota都会累加1。第一行的默认值是0

const (
    A = 10 * iota  // 0 
    B              // 10
    C              // 20
)

# 函数的多返回值

// 多返回值，匿名的
func test(a int, b int) (int, int) {
    return 100, 200
}

// 多返回值，有名字的
func test(a int, b int) (x int, y int) {
    // 此时x与y的作用域在该函数内
    x = 100
    y = 200
    return
}
